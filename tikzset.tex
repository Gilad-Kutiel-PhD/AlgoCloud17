\usetikzlibrary{
	positioning
}

\tikzset{
	on/.code args={<#1>#2}{
  		\only<#1>{\pgfkeysalso{#2}}
	},
	alt/.code args={<#1>#2#3}{
  		\alt<#1>{\pgfkeysalso{#2}}{\pgfkeysalso{#3}}
	},
}

\tikzset{
	label/.style={
		rectangle,	
		draw=none,
		fill=white,
		font=\tiny,
		inner sep=0,
		minimum height=0,
		minimum width=0,
	},
	edge label/.style={
		label,
		midway,
		sloped,
	},
	label below/.style={edge label, below=1mm},
	label above/.style={edge label, above=1mm},
}

\tikzset{
	blur/.style={
		black!20,
		thin
	}
}

	draw, 
	circle,
	inner sep=0mm,
	minimum size=5mm,
	very thick,
	font=\small,
	black!70,
}}

\tikzset{
	>=latex,
	->,

}
